.inputs Button

.outputs Red Yel Green PRed PGreen

.latches
Red -> RedM
Yel -> YelM
Req -> ReqM

.update
Clr = RedM && !YelM
Req = (ReqM || Button) && !Clr
Red = !RedM && YelM || RedM && !YelM
Yel = !RedM && !YelM && Req || RedM && !YelM
Green  = !Red && !Yel
PGreen = Red && !Yel
PRed   = !PGreen

.simulate Button=001001111001
Button=001001111001
Button=001001111001
